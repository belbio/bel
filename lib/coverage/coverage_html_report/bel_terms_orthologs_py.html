18def get_orthologs(term_key: Key, species: List[Key] = []) -> List[dict]: 

19 """Get orthologs for given gene and species 

20 

21 Args: 

22 term_key: gene, rna or protein term key for which to retrieve orthologs 

23 species: target species keys for ortholog - e.g. TAX:<number> 

24 

25 Returns: 

26 List[Term] 

27 """ 

28 

29 canonical_gene_id = bel.terms.terms.get_normalized_terms(term_key)["canonical"] 

30 gene_id_key = bel.db.arangodb.arango_id_to_key(canonical_gene_id) 

31 

32 orthologs = {} 

33 

34 if species: 

35 query_filter = f"FILTER vertex.tax_id IN {species}"

37 query = f""" 

38 LET start = ( 

39 FOR vertex in {ortholog_nodes_coll} 

40 FILTER vertex._key == "{belentity.canonicalize().db_key()}" 

41 RETURN {{ "name": vertex.name, "tax_id": vertex.tax_id }} 

42 ) 

43 

44 LET orthologs = ( 

45 FOR vertex IN 1..3 

46 ANY "ortholog_nodes/{gene_id_key}" ortholog_edges 

47 OPTIONS {{ bfs: true, uniqueVertices : 'global' }} 

48 {query_filter} 

49 RETURN DISTINCT {{ "name": vertex.name, "tax_id": vertex.tax_id }} 

50 ) 

51 

52 RETURN {{ 'orthologs': FLATTEN(UNION(start, orthologs)) }} 

53 """ 

54 

55 cursor = resources_db.aql.execute(query, batch_size=20)

60 orthologs[ortholog["tax_id"]] = {
