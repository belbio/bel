36def convert_csv_str_to_list(csv_str: str) -> list: 

37 """Convert CSV str to list""" 

38 

39 csv_str = re.sub("^\s*{", "", csv_str) 

40 csv_str = re.sub("}\s*$", "", csv_str) 

41 r = csv.reader([csv_str]) 

42 row = list(r)[0] 

43 new = [] 

44 for col in row: 

45 col = re.sub('^\s*"?\s*', "", col) 

46 col = re.sub('\s*"?\s*$', "", col) 

47 new.append(col) 

48 

49 return new

96def split_bel_stmt(stmt: str, line_num) -> tuple: 

97 """Split bel statement into subject, relation, object tuple""" 

98 

99 m = re.match(f"^(.*?\))\s+([a-zA-Z=\->\|:]+)\s+([\w(]+.*?)$", stmt, flags=0) 

100 if m: 

101 return (m.group(1), m.group(2), m.group(3)) 

102 else: 

103 log.info( 

104 f"Could not parse bel statement into components at line number: {line_num} assertion: {stmt}" 

105 ) 

106 return (stmt, None, None)

109def yield_nanopub(assertions, annotations, line_num): 

110 """Yield nanopub object""" 

111 

112 if not assertions:

114 

115 anno = copy.deepcopy(annotations) 

116 

117 evidence = anno.pop("evidence", None) 

118 stmt_group = anno.pop("statement_group", None) 

119 citation = anno.pop("citation", None) 

120 

121 anno_list = []

123 if isinstance(anno[anno_type], (list, tuple)): 

124 for val in anno[anno_type]:

158def process_definition(line, nanopubs_metadata): 

159 """Process DEFINE line in BEL script""" 

160 

161 matches = re.match('DEFINE\s+(\w+)\s+(\w+)\s+AS\s+URL\s+"(.*?)"\s*$', line) 

162 if matches: 

163 def_type = matches.group(1).lower() 

164 if def_type == "namespace": 

165 def_type = "namespaces" 

166 elif def_type == "annotation": 

167 def_type == "annotations" 

168 

169 key = matches.group(2) 

170 val = matches.group(3) 

171 

172 if def_type in nanopubs_metadata: 

173 nanopubs_metadata[def_type][key] = val 

174 else: 

175 nanopubs_metadata[def_type] = {key: val} 

176 

177 matches = re.match("DEFINE\s+(\w+)\s+(\w+)\s+AS\s+LIST\s+{(.*?)}\s*$", line) 

178 if matches: 

179 def_type = matches.group(1).lower() 

180 if def_type == "namespace": 

181 def_type = "namespaces" 

182 elif def_type == "annotation": 

183 def_type == "annotations" 

184 

185 key = matches.group(2) 

186 val = matches.group(3) 

187 vals = convert_csv_str_to_list(val) 

188 

189 if def_type in nanopubs_metadata: 

190 nanopubs_metadata[def_type][key] = vals 

191 else:

219 """Convert annotations into nanopub_bel annotations format"""

231 annotations["citation"] = process_citation(val) 

232 elif key.lower() == "support" or key.lower() == "evidence": 

233 annotations["evidence"] = val 

234 elif re.match("\s*{.*?}", val): 

235 vals = convert_csv_str_to_list(val) 

236 annotations[key] = vals

238 annotations[key] = val

268def preprocess_belscript(lines): 

269 """ Convert any multi-line SET statements into single line SET statements""" 

270 

271 set_flag = False 

272 for line in lines: 

273 if set_flag is False and re.match("SET", line): 

274 set_flag = True 

275 set_line = [line.rstrip()] 

276 # SET following SET 

277 elif set_flag and re.match("SET", line): 

278 yield f"{' '.join(set_line)}\n" 

279 set_line = [line.rstrip()] 

280 # Blank line following SET yields single line SET 

281 elif set_flag and re.match("\s+$", line): 

282 yield f"{' '.join(set_line)}\n" 

283 yield line 

284 set_flag = False 

285 

286 # Append second, third, ... lines to SET 

287 elif set_flag: 

288 set_line.append(line.rstrip()) 

289 else: 

290 yield line

293def parse_belscript(lines): 

294 """Lines from the BELScript - can be an iterator or list

296 yields Nanopubs in nanopubs_bel-1.0.0 format 

297 """

313 # Get rid of trailing comments 

314 line = re.sub("\/\/.*?$", "", line) 

315 

316 line = line.rstrip() 

317 

318 # Collapse continuation lines 

319 while re.search("\\\s*$", line): 

320 line = line.replace("\\", "") + next(lines) 

321 

322 # Process lines ################################# 

323 if re.match("\s*#", line) or re.match("\s*$", line): 

324 # Skip comments and empty lines 

325 continue 

326 elif re.match("SET DOCUMENT", line): 

327 nanopubs_metadata = process_documentline(line, nanopubs_metadata) 

328 elif re.match("DEFINE", line): 

329 nanopubs_metadata = process_definition(line, nanopubs_metadata) 

330 elif re.match("UNSET", line): 

331 

332 # Process any assertions prior to changing annotations 

333 if assertions: 

334 yield yield_nanopub(assertions, annotations, line_num) 

335 assertions = [] 

336 annotations = process_unset(line, annotations) 

337 

338 elif re.match("SET", line): 

339 # Create nanopubs metadata prior to starting BEL Script statements section 

340 if nanopubs_metadata: 

341 yield yield_metadata(nanopubs_metadata) 

342 nanopubs_metadata = {} 

343 

344 # Process any assertions prior to changing annotations 

345 if assertions: 

346 yield yield_nanopub(assertions, annotations, line_num) 

347 assertions = [] 

348 

349 annotations = process_set(line, annotations) 

350 

351 else: 

352 assertions.append(line)
