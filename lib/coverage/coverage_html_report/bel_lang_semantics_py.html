19def validate(bo, error_level: str = "WARNING") -> Tuple[bool, List[Tuple[str, str]]]: 

20 """Semantically validate BEL AST 

21 

22 Add errors and warnings to bel_obj.validation_messages 

23 

24 Error Levels are similar to log levels - selecting WARNING includes both 

25 WARNING and ERROR, selecting ERROR just includes ERROR 

26 

27 Args: 

28 bo: main BEL language object 

29 error_level: return ERRORs only or also WARNINGs 

30 

31 Returns: 

32 Tuple[bool, List[Tuple[str, str]]]: (is_valid, messages) 

33 """ 

34 

35 if bo.ast: 

36 bo = validate_functions(bo.ast, bo) # No WARNINGs generated in this function 

37 if error_level == "WARNING": 

38 bo = validate_arg_values(bo.ast, bo) # validates NSArg and StrArg values 

39 

40 else: 

41 # Don't show general error if more specific error is already added 

42 errors = [error for error in bo.validation_messages if error[0] == "ERROR"] 

43 if not errors: 

44 bo.validation_messages.append(("ERROR", "Invalid BEL Assertion - cannot parse")) 

45 

46 for msg in bo.validation_messages: 

47 if msg[0] == "ERROR": 

48 bo.parse_valid = False 

49 break 

50 

51 return bo

54def validate_functions(ast: BELAst, bo): 

55 """Recursively validate function signatures 

56 

57 Determine if function matches one of the available signatures. Also, 

58 

59 1. Add entity types to AST NSArg, e.g. Abundance, ... 

60 2. Add optional to AST Arg (optional means it is not a 

61 fixed, required argument and needs to be sorted for 

62 canonicalization, e.g. reactants(A, B, C) ) 

63 

64 Args: 

65 bo: bel object 

66 

67 Returns: 

68 bel object 

69 """ 

70 

71 belspec = bo.ast.belspec 

72 

73 if isinstance(ast, Function): 

74 log.debug(f"Validating: {ast.name}, {ast.function_type}, {ast.args}") 

75 function_signatures = belspec["functions"]["signatures"][ast.name]["signatures"] 

76 

77 function_name = ast.name 

78 (valid_function, messages) = check_function_args( 

79 ast.args, function_signatures, function_name 

80 ) 

81 if not valid_function: 

82 message = ", ".join(messages) 

83 bo.validation_messages.append( 

84 ( 

85 "ERROR", 

86 "Invalid BEL Assertion function {} - problem with function signatures: {}".format( 

87 ast.to_string(), message 

88 ), 

89 ) 

90 ) 

91 bo.parse_valid = False 

92 

93 # Recursively process every NSArg by processing BELAst and Functions 

94 if hasattr(ast, "args"): 

95 for arg in ast.args: 

96 validate_functions(arg, bo) 

97 

98 return bo

101def check_function_args(args, signatures, function_name): 

102 """Check function args - return message if function args don't match function signature 

103 

104 Called from validate_functions 

105 

106 We have following types of arguments to validate: 

107 1. Required, position_dependent arguments, e.g. p(HGNC:AKT1), NSArg HGNC:AKT1 is required and must be first argument 

108 2. Optional, position_dependent arguments, e.g. pmod(P, T, 308) - T and 308 are optional and position_dependent 

109 3. Optional, e.g. loc() modifier can only be found once, but anywhere after the position_dependent arguments 

110 4. Multiple, e.g. var(), can have more than one var() modifier in p() function 

111 

112 Args: 

113 args (Union['Function', 'NSArg', 'StrArg']): AST Function arguments 

114 signatures (Mapping[str, Any]): function signatures from spec_dict, may be more than one per function 

115 function_name (str): passed in to improve error messaging 

116 

117 Returns: 

118 Tuple[bool, List[str]]: (function_valid?, list of error messages per signature) 

119 """ 

120 

121 messages = [] 

122 

123 arg_types = [] 

124 for arg in args: 

125 log.info("Arg", arg=arg) 

126 if arg.type == "Function": 

127 arg_types.append((arg.name, "")) 

128 elif arg.type == "NSArg": 

129 arg_types.append((arg.type, arg.value_types)) 

130 elif arg.type == "StrArg": 

131 arg_types.append((arg.type, arg.value)) 

132 log.info(f"Arg_types {arg_types}") 

133 

134 matched_signature_idx = -1 

135 valid_function = False 

136 for sig_argset_idx, sig_argset in enumerate(signatures): 

137 sig_req_args = sig_argset["req_args"] # required position_dependent arguments 

138 sig_pos_args = sig_argset["pos_args"] # optional position_dependent arguments 

139 sig_opt_args = sig_argset["opt_args"] # optional arguments 

140 sig_mult_args = sig_argset["mult_args"] # multiple arguments 

141 

142 log.debug(f"{sig_argset_idx} Req: {sig_req_args}") 

143 log.debug(f"{sig_argset_idx} Pos: {sig_pos_args}") 

144 log.debug(f"{sig_argset_idx} Opt: {sig_opt_args}") 

145 log.debug(f"{sig_argset_idx} Mult: {sig_mult_args}") 

146 

147 # Check required arguments 

148 reqs_mismatch_flag = False 

149 for sig_idx, sig_req in enumerate(sig_req_args): 

150 if len(arg_types) > sig_idx: 

151 log.debug( 

152 "Req args: arg_type {} vs sig_req {}".format(arg_types[sig_idx][0], sig_req) 

153 ) 

154 if arg_types[sig_idx][0] not in sig_req: 

155 reqs_mismatch_flag = True 

156 msg = f"Missing required arguments for {function_name} signature: {sig_argset_idx}" 

157 messages.append(msg) 

158 log.debug(msg) 

159 break 

160 

161 if reqs_mismatch_flag: 

162 continue # test next argset 

163 

164 # Check position_dependent optional arguments 

165 pos_dep_arg_types = arg_types[len(sig_req_args) :] 

166 log.debug(f"Optional arg types {pos_dep_arg_types}") 

167 log.debug(f"{sig_argset_idx} Pos: {sig_pos_args}") 

168 pos_mismatch_flag = False 

169 for sig_pos_idx, sig_pos in enumerate(sig_pos_args): 

170 if sig_pos_idx == len(pos_dep_arg_types): 

171 break # stop checking position dependent arguments when we run out of them vs signature optional position dependent arguments 

172 if pos_dep_arg_types[sig_pos_idx][0] not in sig_pos: 

173 pos_mismatch_flag = True 

174 msg = f"Missing position_dependent arguments for {function_name} signature: {sig_argset_idx}" 

175 messages.append(msg) 

176 log.debug(msg) 

177 break 

178 if pos_mismatch_flag: 

179 continue # test next argset 

180 

181 reqpos_arglen = len(sig_req_args) + len(sig_pos_args) 

182 optional_arg_types = arg_types[reqpos_arglen:] 

183 

184 # Remove function args that are found in the mult_args signature 

185 optional_types = [ 

186 (opt_type, opt_val) 

187 for opt_type, opt_val in optional_arg_types 

188 if opt_type not in sig_mult_args 

189 ] 

190 log.debug(f"Optional types after sig mult args removed {optional_types}") 

191 

192 # Check if any remaining function args are duplicated and therefore not unique opt_args 

193 if len(optional_types) != len(set(optional_types)): 

194 msg = f"Duplicate optional arguments {optional_types} for {function_name} signature: {sig_argset_idx}" 

195 messages.append(msg) 

196 log.debug(msg) 

197 continue 

198 

199 optional_types = [ 

200 (opt_type, opt_val) 

201 for opt_type, opt_val in optional_types 

202 if opt_type not in sig_opt_args 

203 ] 

204 if len(optional_types) > 0: 

205 msg = f"Invalid arguments {optional_types} for {function_name} signature: {sig_argset_idx}" 

206 messages.append(msg) 

207 log.debug(msg) 

208 continue 

209 

210 matched_signature_idx = sig_argset_idx 

211 messages = [] # reset messages if signature is matched 

212 valid_function = True 

213 break 

214 

215 # Add NSArg and StrArg value types (e.g. Protein, Complex, ec) 

216 if matched_signature_idx > -1: 

217 # Shouldn't have single optional NSArg arguments - not currently checking for that 

218 log.debug(f'AST1, Sigs {signatures[matched_signature_idx]["arguments"]} Args: {args}') 

219 for arg_idx, arg in enumerate(args): 

220 log.debug(f"Arg type {arg.type}") 

221 for sig_idx, sig_arg in enumerate(signatures[matched_signature_idx]["arguments"]): 

222 if arg.type == "Function" or sig_arg["type"] in [ 

223 "Function", 

224 "Modifier", 

225 ]: 

226 pass # Skip Function arguments 

227 elif sig_arg.get("position", None): 

228 if sig_arg["position"] == arg_idx + 1: 

229 log.debug("1Adding value_types", arg=arg, value_type=sig_arg["values"]) 

230 arg.add_value_types(sig_arg["values"]) 

231 log.debug(f'AST2 {arg} {sig_arg["values"]}') 

232 elif arg.type in ["NSArg", "StrArg", "StrArgNSArg"]: 

233 log.debug("2Adding value_types", arg=arg, value_type=sig_arg["values"]) 

234 arg.add_value_types(sig_arg["values"]) 

235 log.debug(f'AST2 {arg} {sig_arg["values"]}') 

236 

237 for arg in args: 

238 if arg.__class__.__name__ in ["NSArg", "StrArg"]: 

239 log.debug(f"Arg: {arg.to_string()} Value_types: {arg.value_types}") 

240 

241 return (valid_function, messages)

244def validate_arg_values(ast, bo): 

245 """Recursively validate arg (NSArg and StrArg) values 

246 

247 Check that NSArgs are found in BELbio API and match appropriate entity_type. 

248 Check that StrArgs match their value - either default namespace or regex string 

249 

250 Generate a WARNING if not. 

251 

252 Args: 

253 bo: bel object 

254 

255 Returns: 

256 bel object 

257 """ 

258 

259 log.debug(f"AST: {ast}") 

260 

261 belspec = bo.ast.belspec 

262 

263 # Test NSArg terms 

264 if isinstance(ast, NSArg): 

265 

266 term_id = str(ast.entity) 

267 

268 value_types = ast.value_types 

269 log.debug(f"Value types: {value_types} AST value: {ast.entity.id}") 

270 # Default namespaces are defined in the bel_specification file 

271 if ast.entity.namespace == "DEFAULT": # may use the DEFAULT namespace or not 

272 for value_type in value_types: 

273 default_namespace = [ 

274 ns["name"] for ns in belspec["namespaces"][value_type]["info"] 

275 ] + [ns["abbreviation"] for ns in belspec["namespaces"][value_type]["info"]] 

276 

277 if ast.entity.id in default_namespace: 

278 log.debug("Default namespace valid term: {}".format(term_id)) 

279 break 

280 else: # if for loop doesn't hit the break, run this else 

281 log.debug("Default namespace invalid term: {}".format(term_id)) 

282 bo.validation_messages.append(("WARNING", f"Default Term: {term_id} not found")) 

283 

284 # Process normal, non-default-namespace terms 

285 else: 

286 terms = bel.terms.terms.get_terms(term_id) 

287 if len(terms) > 0: 

288 term = terms[0] 

289 else: 

290 term = {} 

291 

292 # Check that entity types match 

293 if len(set(ast.value_types).intersection(term.get("entity_types", []))) == 0: 

294 log.debug( 

295 f"Invalid Term - Assertion term {term_id} allowable entity types: {ast.value_types} do not match API term entity types: {term.get('entity_types', [])}", 

296 ) 

297 

298 bo.validation_messages.append( 

299 ( 

300 "WARNING", 

301 f"Invalid Term - Assertion term {term_id} allowable entity types: {ast.value_types} do not match API term entity types: {term.get('entity_types', [])}", 

302 ) 

303 ) 

304 

305 if term_id in term.get("obsolete_keys", []): 

306 bo.validation_messages.append( 

307 ("WARNING", f'Obsolete term: {term_id} Current term: {term["id"]}',) 

308 ) 

309 

310 # Process StrArgs 

311 if isinstance(ast, StrArg): 

312 log.debug(f" Check String Arg: {ast.value} {ast.value_types}") 

313 for value_type in ast.value_types: 

314 # Is this a regex to match against 

315 if re.match("/", value_type): 

316 value_type = re.sub("^/", "", value_type) 

317 value_type = re.sub("/$", "", value_type) 

318 match = re.match(value_type, ast.value) 

319 if match: 

320 break 

321 if value_type in belspec["namespaces"]: 

322 default_namespace = [ 

323 ns["name"] for ns in belspec["namespaces"][value_type]["info"] 

324 ] + [ns["abbreviation"] for ns in belspec["namespaces"][value_type]["info"]] 

325 if ast.value in default_namespace: 

326 break 

327 else: # If for loop doesn't hit the break, no matches found, therefore for StrArg value is bad 

328 bo.validation_messages.append( 

329 ( 

330 "WARNING", 

331 f"String value {ast.value} does not match default namespace value or regex pattern: {ast.value_types}", 

332 ) 

333 ) 

334 

335 # Recursively process every NSArg by processing BELAst and Functions 

336 if hasattr(ast, "args"): 

337 for arg in ast.args: 

338 validate_arg_values(arg, bo) 

339 

340 return bo
